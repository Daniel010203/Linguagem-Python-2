#Uma escola solicitou um algoritmo que receba a informação das notas dos alunos
# e mostre a média das mesmas.
#Insersão das notas:
notaMatematica = float(input("Digite a nota na matéria Matemática: "))
notaPortugues = float(input("Digite a nota na matéria Português: "))
notaHistoria = float(input("Digite a nota na matéria História: "))
notaGeografia = float(input("Digite a nota na matéria Geografia: "))
notaLinguaEstrangeira = float(input("Digite a nota na matéria Lingua Estrangueira: "))
notaEducacaoFisica = float(input("Digite a nota na matéria Educação Física: "))

#Operação:
somaDasNotas = notaMatematica + notaPortugues + notaHistoria + notaGeografia + notaLinguaEstrangeira + notaEducacaoFisica
media = somaDasNotas/6
print(" A média do aluno foi de{:.2f}",format(media))
